### 1. TypeScript型定義 🔵

```typescript
export type SourceType = "deck" | "tag";

export interface StudyNotesData {
  source_type: SourceType;
  source_id: string;
  content: string;
  card_count: number;
  is_stale: boolean;
  model_used: string;
  processing_time_ms: number;
  generated_at: string;
}

export interface StudyNotesResponse {
  success: boolean;
  data: StudyNotesData | null;
}
```

**実装ファイル**: `frontend/src/types/studyNotes.ts`

### 2. API サービス関数 🔵

**信頼性**: 🔵 *api-endpoints.md・既存APIサービスパターンより*

```typescript
export async function generateStudyNotes(request: GenerateStudyNotesRequest): Promise<StudyNotesResponse> {
  const response = await apiClient.post("/study-notes/generate", request);
  return response.data;
}

export async function getStudyNotes(sourceType: SourceType, sourceId: string): Promise<StudyNotesResponse> {
  const response = await apiClient.get("/study-notes", {
    params: { source_type: sourceType, source_id: sourceId }
  });
  return response.data;
}
```

**実装ファイル**: `frontend/src/services/studyNotesApi.ts`
