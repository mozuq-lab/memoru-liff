## 実装詳細

### 1. コンポーネント構成 🟡

**信頼性**: 🟡 *REQ-ASN-011・REQ-ASN-201から妥当な推測*

```tsx
export function StudyNotesSection({ sourceType, sourceId }: Props) {
  const { status, data, error, generate, refresh } = useStudyNotes(sourceType, sourceId);

  // status に応じた表示切り替え
  switch (status) {
    case "loading": return <LoadingSkeleton />;
    case "generating": return <GeneratingIndicator />;
    case "cached": return <NotesContent content={data.content} />;
    case "stale": return <StaleNotice data={data} onRegenerate={generate} />;
    case "empty": return <EmptyState onGenerate={generate} />;
    case "error": return <ErrorState message={error} onRetry={generate} />;
  }
}
```

**実装ファイル**: `frontend/src/components/StudyNotesSection.tsx`
