### 1. エンドポイント実装

#### File: `backend/src/api/handler.py` (追加)

```python
from aws_lambda_powertools import Logger
from models.grading import GradeAnswerRequest, GradeAnswerResponse
from services.exceptions import (
    AITimeoutError,
    AIRateLimitError,
    AIParseError,
    AIInternalError
)

logger = Logger()

@app.post("/reviews/<card_id>/grade-ai")
@validate_json
def grade_answer(card_id: str):
    """
    POST /reviews/{card_id}/grade-ai

    AI を使用してユーザーの回答を採点する

    Request:
    {
        "user_answer": "string (1-2000文字)"
    }

    Response:
    {
        "grade": 0-5,
        "reasoning": "string",
        "card_front": "string",
        "card_back": "string",
        "grading_info": {
            "model_used": "strands" or "bedrock",
            "processing_time_ms": int
        }
    }

    Status Codes:
    - 200: 採点成功
    - 400: バリデーションエラー（回答が空、形式不正）
    - 401: 認証エラー
    - 404: カードが見つからない / 他ユーザーのカード
    - 429: AI レート制限超過
    - 500: AI パースエラー / 内部エラー
    - 503: AI プロバイダーエラー
    - 504: AI タイムアウト
    """
    try:
        # ユーザー認証
        user_id = get_user_id_from_jwt()
        if not user_id:
            return make_response({"error": "Unauthorized"}, 401)

        # リクエストバリデーション
        try:
            request_data = GradeAnswerRequest(**request.json)
        except ValueError as e:
            logger.warning(f"Validation error: {str(e)}", extra={
                "card_id": card_id,
                "user_id": user_id
            })
            return make_response({
                "error": "Bad Request",
                "detail": str(e)
            }, 400)

        # カード取得（ユーザー認可チェック）
        card = card_service.get_card(user_id, card_id)
        if not card:
            logger.warning(f"Card not found or unauthorized", extra={
                "card_id": card_id,
                "user_id": user_id
            })
            return make_response({"error": "Not Found"}, 404)

        # card_front, card_back を取得
        card_front = card.get("front")
        card_back = card.get("back")

        logger.info(f"Grading answer for card", extra={
            "card_id": card_id,
            "user_id": user_id,
            "user_answer_length": len(request_data.user_answer)
        })

        # AI 採点呼び出し
        language = request.args.get("language", "ja")
        grading_result = ai_service.grade_answer(
            card_front=card_front,
            card_back=card_back,
            user_answer=request_data.user_answer,
            language=language
        )

        # レスポンス構築
        response_data = GradeAnswerResponse(
            grade=grading_result.grade,
            reasoning=grading_result.reasoning,
            card_front=card_front,
            card_back=card_back,
            grading_info={
                "model_used": grading_result.model_used,
                "processing_time_ms": grading_result.processing_time_ms
            }
        )

        logger.info(f"Answer graded successfully", extra={
            "card_id": card_id,
            "user_id": user_id,
            "grade": grading_result.grade
        })

        return make_response(response_data.model_dump(), 200)

    except AITimeoutError as e:
        logger.error(f"AI timeout: {str(e)}", extra={
            "card_id": card_id,
            "user_id": user_id
        })
        return make_response({"error": "Gateway Timeout"}, 504)

    except AIRateLimitError as e:
        logger.warning(f"AI rate limit exceeded: {str(e)}", extra={
            "card_id": card_id,
            "user_id": user_id
        })
        return make_response({"error": "Too Many Requests"}, 429)

    except AIParseError as e:
        logger.error(f"AI parse error: {str(e)}", extra={
            "card_id": card_id,
            "user_id": user_id
        })
        return make_response({"error": "Internal Server Error"}, 500)

    except AIInternalError as e:
        logger.error(f"AI internal error: {str(e)}", extra={
            "card_id": card_id,
            "user_id": user_id
        })
        return make_response({"error": "Internal Server Error"}, 500)

    except Exception as e:
        logger.exception(f"Unexpected error: {str(e)}", extra={
            "card_id": card_id,
            "user_id": user_id
        })
        return make_response({"error": "Internal Server Error"}, 500)
```

### 2. エラーハンドリング詳細

| エラーコード | HTTP Status | 説明 |
|-----------|------------|------|
| ValidationError | 400 | 回答が空、2000文字超、形式不正 |
| CardNotFound | 404 | カードが見つからない |
| UnauthorizedCard | 404 | 他ユーザーのカード |
| AITimeoutError | 504 | AI タイムアウト（30秒以上） |
| AIRateLimitError | 429 | AI レート制限超過 |
| AIParseError | 500 | JSON パースエラー |
| AIInternalError | 500 | その他の AI エラー |
| AIProviderError | 503 | Bedrock/Strands プロバイダーエラー |

### 3. ロギング仕様

すべてのリクエストについて以下をログ出力:
- `user_id`: ユーザーID
- `card_id`: カードID
- `user_answer_length`: 回答の文字数
- `grade`: 採点結果（成功時）
- `processing_time_ms`: AI 処理時間（成功時）
- `model_used`: 使用 AI モデル（strands / bedrock）
