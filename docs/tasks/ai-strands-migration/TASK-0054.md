### 4. `prompts/grading.py` - 回答採点プロンプト（新規）

```python
"""回答採点プロンプト（SM-2 グレード 0-5）"""

from typing import Literal

Language = Literal["ja", "en"]

SM2_GRADE_DEFINITIONS = """
SM-2 採点グレード定義（0-5段階）:

5: 完璧な回答
   - 完全に正確で詳細
   - 補足情報を含む

4: 良好な回答
   - 正確で完全な理解を示す
   - わずかな不正確さなし

3: 許容可能な回答
   - 基本的に正確
   - わずかな重要な詳細が欠けている

2: 弱い回答
   - 基本的な理解はあるが、重要な詳細が欠けている
   - いくつかの不正確さあり

1: 非常に弱い回答
   - わずかな理解のみ
   - 多くの誤解または欠落あり

0: 完全に不正確または回答なし
"""

GRADING_SYSTEM_PROMPT = f"""
You are an expert tutor evaluating student answers using the SM-2 spaced repetition system.

{SM2_GRADE_DEFINITIONS}

Grade the student's answer on the question according to these criteria.
Consider:
1. Accuracy
2. Completeness
3. Understanding of key concepts
4. Any misconceptions

Respond with a JSON object containing:
{{
  "grade": <0-5>,
  "reasoning": "<brief explanation in target language>",
  "feedback": "<constructive feedback for the student>"
}}
"""

def get_grading_prompt(
    card_front: str,
    card_back: str,
    user_answer: str,
    language: Language = "ja"
) -> str:
    """
    回答採点プロンプトを生成する

    Args:
        card_front: カード表（質問）
        card_back: カード裏（正解）
        user_answer: ユーザー回答
        language: 言語（"ja" または "en"）

    Returns:
        プロンプト文字列
    """
    lang_instruction = {
        "ja": "Respond in Japanese.",
        "en": "Respond in English.",
    }.get(language, "Respond in Japanese.")

    prompt = f"""
Question (Card Front):
{card_front}

Correct Answer (Card Back):
{card_back}

Student's Answer:
{user_answer}

{lang_instruction}

Grade this answer and provide detailed feedback for the student.
"""
    return prompt
```

### 5. `prompts/advice.py` - 学習アドバイスプロンプト（新規）

```python
"""学習アドバイスプロンプト"""

from typing import Literal
from dataclasses import dataclass

Language = Literal["ja", "en"]

ADVICE_SYSTEM_PROMPT = """
You are an expert learning coach providing personalized study advice.

Based on the student's review statistics, provide:
1. Specific, actionable advice to improve weak areas
2. Identification of recurring patterns or misconceptions
3. Recommended study strategies for improvement

Respond with a JSON object containing:
{
  "advice_text": "<main personalized advice>",
  "weak_areas": ["<area1>", "<area2>", ...],
  "recommendations": ["<strategy1>", "<strategy2>", ...]
}
"""

def get_advice_prompt(
    review_summary: dict,
    language: Language = "ja"
) -> str:
    """
    学習アドバイスプロンプトを生成する

    Args:
        review_summary: 復習統計サマリー（dict または ReviewSummary）
        language: 言語（"ja" または "en"）

    Returns:
        プロンプト文字列
    """
    # ReviewSummary または dict から情報を抽出
    if hasattr(review_summary, '__dict__'):
        stats = review_summary.__dict__
    else:
        stats = review_summary

    lang_instruction = {
        "ja": "Respond in Japanese.",
        "en": "Respond in English.",
    }.get(language, "Respond in Japanese.")

    stats_text = f"""
Student Review Statistics:
- Total Reviews: {stats.get('total_reviews', 0)}
- Average Grade (0-5): {stats.get('average_grade', 0):.1f}
- Total Cards: {stats.get('total_cards', 0)}
- Cards Due Today: {stats.get('cards_due_today', 0)}
- Streak Days: {stats.get('streak_days', 0)}
- Tag Performance: {stats.get('tag_performance', {})}
"""

    prompt = f"""
{stats_text}

Based on this student's learning statistics, provide personalized advice.
{lang_instruction}

Focus on:
1. Areas where the student is struggling (average grade < 3)
2. Patterns in their review history
3. Specific recommendations to improve
"""
    return prompt
```
