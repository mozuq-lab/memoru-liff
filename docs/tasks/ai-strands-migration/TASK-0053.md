### 4. ファクトリ関数

```python
def create_ai_service(use_strands: bool = None) -> AIService:
    """
    AI サービスインスタンスを作成する

    Args:
        use_strands: Strands SDK を使用するかどうか。
                    None の場合は USE_STRANDS 環境変数を参照

    Returns:
        AIService Protocol に準拠したインスタンス

    Raises:
        AIProviderError: SDK の初期化に失敗した場合
    """
    import os
    from services.bedrock import BedrockAIService
    from services.strands import StrandsAIService

    if use_strands is None:
        use_strands = os.getenv("USE_STRANDS", "false").lower() == "true"

    try:
        if use_strands:
            return StrandsAIService()
        else:
            return BedrockAIService()
    except Exception as e:
        raise AIProviderError(f"Failed to initialize AI service: {e}")
```
