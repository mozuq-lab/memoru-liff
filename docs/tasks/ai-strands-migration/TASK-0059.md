from pydantic import BaseModel, field_validator
from typing import Optional, Dict, Any
from datetime import datetime

class GradeAnswerRequest(BaseModel):
    """回答採点リクエストモデル"""
    user_answer: str

    @field_validator('user_answer')
    @classmethod
    def validate_user_answer(cls, v):
        """
        - 1-2000文字の範囲内
        - 空白のみ（スペース/タブ/改行のみ）は不可
        """
        if not v or len(v.strip()) == 0:
            raise ValueError("user_answer cannot be empty or whitespace only")
        if len(v) > 2000:
            raise ValueError("user_answer must be 2000 characters or less")
        return v

    class Config:
        json_schema_extra = {
            "example": {
                "user_answer": "The capital of France is Paris"
            }
        }

class GradeAnswerResponse(BaseModel):
    """回答採点レスポンスモデル"""
    grade: int  # 0-5: SM-2グレード
    reasoning: str  # 採点理由
    card_front: str  # 質問（参考表示用）
    card_back: str  # 正解（参考表示用）
    grading_info: Dict[str, Any]  # {model_used, processing_time_ms}

    class Config:
        json_schema_extra = {
            "example": {
                "grade": 4,
                "reasoning": "Correct answer with minor hesitation in phrasing",
                "card_front": "What is the capital of France?",
                "card_back": "Paris",
                "grading_info": {
                    "model_used": "strands",
                    "processing_time_ms": 1234
                }
            }
        }
```

from typing import Optional

# SM-2 グレード定義
SM2_GRADES = {
    0: "Complete blackout (no memory of the fact)",
    1: "Incorrect response, but upon reflection know where the error is",
    2: "Incorrect response, but easy to recall the correct answer",
    3: "Correct response, but with serious doubt",
    4: "Correct response after a hesitation",
    5: "Perfect response"
}

GRADING_SYSTEM_PROMPT = """You are an expert Japanese language teacher and assessment specialist.
Your task is to grade student responses based on the quality of their answer.

You will evaluate answers using the SM-2 spaced repetition grading scale (0-5):
- 0: Complete blackout (no memory of the fact)
- 1: Incorrect response, but upon reflection know where the error is
- 2: Incorrect response, but easy to recall the correct answer
- 3: Correct response, but with serious doubt
- 4: Correct response after a hesitation
- 5: Perfect response

Consider:
- Correctness of the answer
- Completeness
- Clarity
- Grammar and language quality (for language learning)
- Minor spelling/punctuation errors (still 4-5)
- Conceptual understanding (for knowledge cards)

Always respond with valid JSON in this exact format:
{
    "grade": <0-5>,
    "reasoning": "<brief explanation of grade decision>"
}"""

def get_grading_prompt(
    card_front: str,
    card_back: str,
    user_answer: str,
    language: str = "ja"
) -> str:
    """
    生成: 回答採点プロンプト

    Args:
        card_front: カードの表（質問）
        card_back: カードの裏（正解）
        user_answer: ユーザーの回答
        language: 言語 ("ja" or "en")

    Returns:
        構成されたプロンプト文字列
    """
    if language == "en":
        instruction = """Grade this student response on the SM-2 scale (0-5).
Return valid JSON with "grade" and "reasoning" fields."""
    else:
        instruction = """学生の回答をSM-2スケール(0-5)で採点してください。
"grade"と"reasoning"フィールドを含む有効なJSONで返してください。"""

    prompt = f"""{instruction}

**Question (Card Front):**
{card_front}

**Expected Answer (Card Back):**
{card_back}

**Student's Response:**
{user_answer}

Grade this response and explain your decision."""

    return prompt
```

    Returns:
        GradingResult (grade, reasoning)

    Raises:
        AITimeoutError: タイムアウト
