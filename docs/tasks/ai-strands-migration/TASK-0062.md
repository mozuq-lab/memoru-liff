```python
from models.advice import ReviewSummary, LearningAdviceResponse
from services.prompts.advice import get_advice_prompt, ADVICE_SYSTEM_PROMPT
import json
import asyncio
import time

class StrandsAIService:
    """既存メソッドに加えて以下を追加"""

    async def get_learning_advice(
        self,
        review_summary: ReviewSummary,
        language: str = "ja"
    ) -> LearningAdviceResult:
        """
        Strands Agents SDK を使用して学習アドバイスを生成

        Args:
            review_summary: ReviewSummary オブジェクト（学習データ集計）
            language: 言語 ("ja" or "en")

        Returns:
            LearningAdviceResult (advice_text, weak_areas, recommendations)

        Raises:
            AITimeoutError: タイムアウト
            AIRateLimitError: レート制限超過
            AIParseError: JSON パースエラー
            AIInternalError: その他の AI エラー
        """
        try:
            start_time = time.time()

            # プロンプト構築（ReviewSummary を埋め込む）
            prompt = get_advice_prompt(review_summary, language)

            # Strands Agent 作成・実行
            agent = self.client.agents.create(
                name="learning_advice_agent",
                system_prompt=ADVICE_SYSTEM_PROMPT
            )

            response = await asyncio.wait_for(
                self._invoke_agent(agent, prompt),
                timeout=self.timeout
            )

            processing_time_ms = int((time.time() - start_time) * 1000)

            # JSON パース
            result = self._parse_advice_result(response, processing_time_ms)
            return result

        except asyncio.TimeoutError:
            raise AITimeoutError("Strands learning advice request timed out")
        except (json.JSONDecodeError, ValueError) as e:
            raise AIParseError(f"Failed to parse advice response: {str(e)}")
        # ... その他のエラーハンドリング

    def _parse_advice_result(self, response: str, processing_time_ms: int) -> LearningAdviceResult:
        """
        Agent レスポンスを LearningAdviceResult に変換

        期待形式:
        {
            "advice_text": "...",
            "weak_areas": ["area1", "area2"],
            "recommendations": ["rec1", "rec2"]
        }
        """
        data = json.loads(response)
        return LearningAdviceResult(
            advice_text=data["advice_text"],
            weak_areas=data.get("weak_areas", []),
            recommendations=data.get("recommendations", []),
            processing_time_ms=processing_time_ms,
            model_used="strands"
        )
```

### 2. エンドポイント実装

```python
from aws_lambda_powertools import Logger
from models.advice import ReviewSummary, LearningAdviceResponse
from services.review_service import ReviewService
from services.exceptions import (
    AITimeoutError,
    AIRateLimitError,
    AIParseError,
    AIInternalError
)

logger = Logger()

@app.get("/advice")
def get_advice():
    """
    GET /advice

    ユーザーの学習データを集計し、AI が学習アドバイスを生成する

    Query Parameters:
    - language: 言語 ("ja" or "en")、デフォルト "ja"

    Response:
    {
        "advice_text": "string",
        "weak_areas": ["string"],
        "recommendations": ["string"],
        "study_stats": {
            "total_reviews": int,
            "average_grade": float,
            "streak_days": int,
            "cards_due_today": int
        },
        "advice_info": {
            "model_used": "strands" or "bedrock",
            "processing_time_ms": int
        }
    }

    Status Codes:
    - 200: アドバイス取得成功
    - 401: 認証エラー
    - 429: AI レート制限超過
    - 500: DynamoDB アクセスエラー / AI エラー
    - 503: AI プロバイダーエラー
    - 504: AI タイムアウト
    """
    try:
        # ユーザー認証
        user_id = get_user_id_from_jwt()
        if not user_id:
            return make_response({"error": "Unauthorized"}, 401)

        # 言語パラメータ取得
        language = request.args.get("language", "ja")
        if language not in ["ja", "en"]:
            language = "ja"

        logger.info(f"Getting learning advice for user", extra={
            "user_id": user_id,
            "language": language
        })

        # 1. ReviewService で学習データを集計
        review_summary = review_service.get_review_summary(user_id)

        # 2. AI で学習アドバイスを生成
        advice_result = ai_service.get_learning_advice(
            review_summary=review_summary,
            language=language
        )

        # 3. レスポンス構築
        study_stats = {
            "total_reviews": review_summary.total_reviews,
            "average_grade": review_summary.average_grade,
            "streak_days": review_summary.streak_days,
            "cards_due_today": review_summary.cards_due_today
        }

        response_data = LearningAdviceResponse(
            advice_text=advice_result.advice_text,
            weak_areas=advice_result.weak_areas,
            recommendations=advice_result.recommendations,
            study_stats=study_stats,
            advice_info={
                "model_used": advice_result.model_used,
                "processing_time_ms": advice_result.processing_time_ms
            }
        )

        logger.info(f"Learning advice generated successfully", extra={
            "user_id": user_id,
            "weak_areas_count": len(advice_result.weak_areas),
            "recommendations_count": len(advice_result.recommendations),
            "total_reviews": review_summary.total_reviews
        })

        return make_response(response_data.model_dump(), 200)

    except AITimeoutError as e:
        logger.error(f"AI timeout: {str(e)}", extra={
            "user_id": user_id
        })
        return make_response({"error": "Gateway Timeout"}, 504)

    except AIRateLimitError as e:
        logger.warning(f"AI rate limit exceeded: {str(e)}", extra={
            "user_id": user_id
        })
        return make_response({"error": "Too Many Requests"}, 429)

    except (AIParseError, AIInternalError) as e:
        logger.error(f"AI error: {str(e)}", extra={
            "user_id": user_id
        })
        return make_response({"error": "Internal Server Error"}, 500)

    except Exception as e:
        logger.exception(f"Unexpected error: {str(e)}", extra={
            "user_id": user_id
        })
        return make_response({"error": "Internal Server Error"}, 500)
```

### 3. エラーハンドリング詳細

| エラーコード | HTTP Status | 説明 |
|-----------|------------|------|
| AITimeoutError | 504 | AI タイムアウト（30秒以上） |
| AIRateLimitError | 429 | AI レート制限超過 |
| AIParseError | 500 | JSON パースエラー |
| AIInternalError | 500 | その他の AI エラー |
| AIProviderError | 503 | Bedrock/Strands プロバイダーエラー |
| DynamoDB エラー | 500 | DynamoDB アクセスエラー |

### 4. ロギング仕様

すべてのリクエストについて以下をログ出力:
- `user_id`: ユーザーID
- `language`: 指定言語（ja/en）
- `total_reviews`: 総復習回数
- `weak_areas_count`: 弱点分野の数（成功時）
- `recommendations_count`: 推奨アクションの数（成功時）
- `processing_time_ms`: AI 処理時間（成功時）
- `model_used`: 使用 AI モデル（strands / bedrock）
