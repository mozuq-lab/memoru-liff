```python
import os
from aws_lambda_powertools import Logger

logger = Logger()

_agentcore_client = None

def _get_agentcore_client():
    global _agentcore_client
    if _agentcore_client is None:
        from bedrock_agentcore.memory import AgentCoreMemoryClient
        _agentcore_client = AgentCoreMemoryClient()
    return _agentcore_client

def create_tutor_session_manager(session_id: str, user_id: str, backend: str | None = None):
    if backend is None:
        backend = os.environ.get("TUTOR_SESSION_BACKEND", "")
        if not backend:
            env = os.environ.get("ENVIRONMENT", "prod")
            backend = "dynamodb" if env == "dev" else "agentcore"

    if backend == "agentcore":
        # AgentCore Memory バックエンド
        ...
    elif backend == "dynamodb":
        # DynamoDB バックエンド
        ...
    else:
        raise ValueError(f"Unknown TUTOR_SESSION_BACKEND: {backend}")
```

### 2. バックエンド自動選択ロジック 🔵

**信頼性**: 🔵 *REQ-103, REQ-104・ユーザヒアリングより*

- `TUTOR_SESSION_BACKEND` 設定済み → その値を使用
- 未設定 + `ENVIRONMENT=dev` → `dynamodb`
- 未設定 + `ENVIRONMENT=prod/staging` → `agentcore`
